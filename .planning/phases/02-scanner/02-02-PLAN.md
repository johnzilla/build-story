```typescript
import type { ArtifactSource } from '../types/source.js'
import type { ScanOptions } from '../types/options.js'

export const DEFAULT_PATTERNS: string[] = [
  // GStack artifacts (SCAN-02)
  'PLANNING.md', 'PLAN.md', 'ARCHITECTURE.md', 'DECISIONS.md',
  'ROADMAP.md', 'STATUS.md', 'CHANGELOG.md',
  '*.gstack', '.gstack/**/*.md',
  // GSD artifacts (SCAN-03)
  'TASKS.md', 'TODO.md', 'SESSION_LOG.md', 'BLOCKERS.md',
  '*.gsd', '.gsd/**/*.md',
  '.planning/**/*.md',
  '.claude/**/*.md',
  // Generic planning artifacts (SCAN-04)
  'ADR/**/*.md', 'adr/**/*.md',
  'docs/**/*.md',
  'README.md',
]

export const DEFAULT_EXCLUDES: string[] = [
  'node_modules/**', '.git/**', 'vendor/**',
  'dist/**', 'target/**', 'build/**', '.turbo/**',
]

export async function discoverFiles(
  source: ArtifactSource,
  options: ScanOptions,
): Promise<string[]> {
  const patterns = options.patterns ?? DEFAULT_PATTERNS
  const excludes = [
    ...DEFAULT_EXCLUDES,
    ...(options.excludes ?? []),
  ]
  return source.glob(patterns, {
    cwd: options.rootDir,
    ignore: excludes,
  })
}
```

Note: `maxDepth` is a hint for the glob implementation (fast-glob's `deep` option). The core file-walker passes it through via the glob options. However, ArtifactSource.glob() currently only accepts `{ cwd?, ignore? }`. The executor should add `maxDepth?: number` to the glob options type in source.ts if not already present, OR pass maxDepth via a convention (the CLI glob adapter reads it). The simplest approach: add `maxDepth?: number` to the glob options interface in source.ts during this task. Then the file-walker passes `{ cwd, ignore, maxDepth: options.maxDepth ?? 5 }`.

**1. Event ID generation function:**

```typescript
export function generateEventId(
  source: 'file' | 'git-commit' | 'git-tag',
  path: string,
  date: string,
): string {
  const input = `${source}:${path}:${date}`
  // djb2 hash — deterministic, no crypto import needed
  let hash = 5381
  for (let i = 0; i < input.length; i++) {
    hash = ((hash << 5) + hash) ^ input.charCodeAt(i)
    hash = hash >>> 0 // force unsigned 32-bit
  }
  const prefix = source === 'file' ? 'file' : source === 'git-tag' ? 'tag' : 'commit'
  return `${prefix}-${hash.toString(16).padStart(8, '0')}`
}
```
