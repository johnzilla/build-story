**packages/cli/src/config.ts:** Per D-08, load TOML from project root and ~/.config/buildstory/config.toml. Project overrides global. Per D-09, never read API keys from config — they come from env only. Wrap smol-toml parse() in try/catch for malformed TOML (per PITFALLS.md security threat pattern).
```typescript
import { readFileSync, existsSync } from 'fs'
import { join } from 'path'
import { homedir } from 'os'
import { parse } from 'smol-toml'

export interface BuildStoryConfig {
  provider?: 'anthropic' | 'openai'
  style?: 'technical' | 'overview' | 'retrospective' | 'pitch'
  outputDir?: string
  scan?: {
    patterns?: string[]
    excludes?: string[]
    maxDepth?: number
  }
}

export function loadConfig(projectRoot: string): BuildStoryConfig {
  const globalPath = join(homedir(), '.config', 'buildstory', 'config.toml')
  const projectPath = join(projectRoot, 'buildstory.toml')

  let globalConfig: BuildStoryConfig = {}
  if (existsSync(globalPath)) {
    try {
      globalConfig = parse(readFileSync(globalPath, 'utf8')) as BuildStoryConfig
    } catch (err) {
      console.error(`Warning: Failed to parse global config at ${globalPath}: ${(err as Error).message}`)
    }
  }

  let projectConfig: BuildStoryConfig = {}
  if (existsSync(projectPath)) {
    try {
      projectConfig = parse(readFileSync(projectPath, 'utf8')) as BuildStoryConfig
    } catch (err) {
      console.error(`Warning: Failed to parse project config at ${projectPath}: ${(err as Error).message}`)
    }
  }

  return { ...globalConfig, ...projectConfig }
}
```

**packages/cli/src/commands/run.ts:** Per D-11, sequential await. Per D-06, ora spinner per phase. Imports scan/narrate/render from @buildstory/core.
```typescript
import { scan, narrate, render } from '@buildstory/core'
import { loadConfig } from '../config.js'
import ora from 'ora'

export async function run(paths: string[], opts: {
  config?: string
  provider: string
  style: string
  output: string
}) {
  const config = loadConfig(process.cwd())
  const rootDir = paths[0] ?? process.cwd()

  const spinner = ora('Scanning artifacts...').start()
  const timeline = await scan({
    rootDir,
    patterns: config.scan?.patterns,
    excludes: config.scan?.excludes,
    maxDepth: config.scan?.maxDepth,
  })
  spinner.succeed('Scan complete')

  spinner.start('Narrating...')
  const script = await narrate(timeline, {
    provider: (config.provider ?? opts.provider) as 'anthropic' | 'openai',
    style: (config.style ?? opts.style) as 'technical' | 'overview' | 'retrospective' | 'pitch',
    apiKey: process.env.ANTHROPIC_API_KEY ?? process.env.OPENAI_API_KEY ?? '',
  })
  spinner.succeed('Narration complete')

  spinner.start('Rendering...')
  const result = await render(script, {
    outputPath: config.outputDir ?? opts.output,
  })
  spinner.succeed(`Render complete: ${result.videoPath}`)
}
```

**packages/cli/src/__tests__/config.test.ts:** Tests for loadConfig function. Use tmp directories to create test TOML files. Verify:
- Empty return when no files
- Correct parsing of valid TOML with provider and style fields
- Project config overrides global config
- Malformed TOML logs warning to stderr but does not throw

```typescript
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import { mkdtempSync, writeFileSync, mkdirSync, rmSync } from 'fs'
import { join } from 'path'
import { tmpdir } from 'os'
import { loadConfig } from '../config.js'

describe('loadConfig', () => {
  let tmpDir: string

  beforeEach(() => {
    tmpDir = mkdtempSync(join(tmpdir(), 'buildstory-test-'))
  })

  afterEach(() => {
    rmSync(tmpDir, { recursive: true, force: true })
  })

  it('returns empty object when no config files exist', () => {
    const config = loadConfig(tmpDir)
    expect(config).toEqual({})
  })

  it('parses project buildstory.toml', () => {
    writeFileSync(join(tmpDir, 'buildstory.toml'), 'provider = "openai"\nstyle = "technical"\n')
    const config = loadConfig(tmpDir)
    expect(config.provider).toBe('openai')
    expect(config.style).toBe('technical')
  })

  it('handles malformed TOML without throwing', () => {
    const spy = vi.spyOn(console, 'error').mockImplementation(() => {})
    writeFileSync(join(tmpDir, 'buildstory.toml'), '{{invalid toml')
    const config = loadConfig(tmpDir)
    expect(config).toEqual({})
    expect(spy).toHaveBeenCalled()
    spy.mockRestore()
  })
})
```
