export async function detectHeyGenPackage(): Promise<boolean> {
  try {
    await import('@buildstory/heygen')
    return true
  } catch {
    return false
  }
}

export async function ensureHeyGenPackage(): Promise<void> {
  const installed = await detectHeyGenPackage()
  if (installed) return

  const proceed = await askYesNo(
    'HeyGen renderer requires installing @buildstory/heygen. Install now? [Y/n] ',
  )
  if (!proceed) {
    console.log('Skipping HeyGen install.')
    process.exit(0)
  }

  console.log('Installing @buildstory/heygen...')
  const result = spawnSync('pnpm', ['install', '--filter', '@buildstory/heygen'], {
    stdio: 'inherit',
    cwd: process.cwd(),
  })
  if (result.status !== 0) {
    console.error('Failed to install @buildstory/heygen.')
    process.exit(1)
  }
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI flag -> renderer dispatch | User-supplied --renderer value selects code path; only 'heygen' and 'remotion' are valid |
| process.env -> HeyGen options | HEYGEN_API_KEY from environment passed to preflight; empty string triggers failure |
| Config TOML -> HeyGen options | avatarId/voiceId from buildstory.toml; validated by preflight for non-empty |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-04 | Spoofing | render.ts | mitigate | API key read only from `process.env['HEYGEN_API_KEY']` -- never from config files where it could be committed to git. Preflight validates non-empty before use. |
| T-05-05 | Elevation of Privilege | render.ts | accept | Invalid --renderer value (not 'heygen' or 'remotion') falls through to Remotion path -- safe default. No code injection vector. |
| T-05-06 | Information Disclosure | render.ts | accept | Cost estimate output shows avatar_id and voice_id from config -- these are not secrets. API key is never printed. |
</threat_model>
